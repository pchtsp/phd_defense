\begin{frame}
\frametitle{\textbf{Maintenance planning}}

  \begin{columns}[c]
    \column{0.5\linewidth}
    Three types:
    \pause
    \begin{itemize}[<+->]
      \setbeamercolor{alerted text}{fg=black} %change the font color

      \item \alert<5->{Preventive}.
      \item Predictive.
      % \only<1->{\item Corrective.}
      % \only<2-4>{\item Preventive.}
      % \onslide<5->{\item \textbf{Preventive}.}
      % \onslide<4->{\item Predictive.}
    \end{itemize} 

  \pause
  \column{0.5\linewidth}
  Three elements:

  \begin{itemize}[<+->]
    \item Tasks (planned over time).
    \item Resources (to do the tasks).
    \item Recovery tasks (to care for resources).
  \end{itemize}
  \end{columns}

\end{frame}

\begin{frame}
\frametitle{\textbf{Maintenance planning applications}}

  In production:
  \begin{itemize}[<+->]
    \item Tasks: scheduled jobs, product demand.
    \item Resources: machines, factories.
    \item Recovery tasks: setup times, refueling, maintenances.
  \end{itemize}  
  % \pause
  \onslide<5->{In transport: }
  \pause
  \begin{itemize}[<+->]
    \item Tasks: flights, train schedules, routes.
    \item Resources: aircraft, trains, vehicles.
    \item Recovery tasks: checks.
  \end{itemize}

  \pause
  Aircraft: Flight and Maintenance Planning (FMP) problem.

  \pause
  Military FMP.

\end{frame}
