  \ganttsetendanchor{on bottom=0}
  \draw[dashed]
  % % first segment (down)
  (\xLeft, \yUpper-0.2cm) --
  % second segment (right)
  (\xLeft, \yUpper -0.5cm) --
  % link label
  node [pos=.5, anchor=north] {\ganttlinklabel}
  % third segment (up)
  (\xRight, \yLower -0.5cm)
  --  (\xRight, \yUpper -0.2cm);
}
